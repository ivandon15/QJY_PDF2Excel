**Goal:** PyQt6 desktop app that extracts Bilibili video metadata from browser-screenshot PDFs and exports to Excel.

**Architecture:** PyMuPDF extracts raw JPEGs from PDF pages; user draws a reusable crop region on a sample page via PyQt6 canvas; cropped images are sent to Claude haiku vision API; results are written to Excel with openpyxl.

**Tech Stack:** Python 3, PyQt6, PyMuPDF (fitz), anthropic SDK, openpyxl

```python
    def _open_pdf(self):
        path, _ = QFileDialog.getOpenFileName(
            self, "选择 PDF", "files", "PDF Files (*.pdf)"
        )
        if not path:
            return
        self.pdf_path = path
        self.pdf_doc = fitz.open(path)
        self._load_thumbnails()

    def _load_thumbnails(self):
        self.thumb_list.clear()
        for i in range(len(self.pdf_doc)):
            page = self.pdf_doc[i]
            imgs = page.get_images(full=True)
            if imgs:
                xref = imgs[0][0]
                img_data = self.pdf_doc.extract_image(xref)
                qimg = QImage.fromData(img_data["image"])
                pixmap = QPixmap.fromImage(qimg).scaled(
                    160, 90, Qt.AspectRatioMode.KeepAspectRatio,
                    Qt.TransformationMode.SmoothTransformation
                )
                item = QListWidgetItem(f"第 {i+1} 页")
                item.setIcon(pixmap)  # type: ignore
                item.setData(Qt.ItemDataRole.UserRole, i)
                self.thumb_list.addItem(item)

    def _on_thumb_click(self, item):
        page_idx = item.data(Qt.ItemDataRole.UserRole)
        imgs = self.pdf_doc[page_idx].get_images(full=True)
        if imgs:
            xref = imgs[0][0]
            img_data = self.pdf_doc.extract_image(xref)
            qimg = QImage.fromData(img_data["image"])
            self.annotation_view.set_image(qimg)
            self.tabs.setCurrentIndex(0)
```

```python
import json
import os
from PyQt6.QtWidgets import QWidget, QVBoxLayout, QHBoxLayout, QPushButton, QLabel, QScrollArea
from PyQt6.QtGui import QPixmap, QImage, QPainter, QPen, QColor
from PyQt6.QtCore import Qt, QRect, QPoint

REGIONS_FILE = "regions.json"

class AnnotationView(QWidget):
    def __init__(self):
        super().__init__()
        self.image = None          # QImage
        self.pixmap = None         # scaled QPixmap for display
        self.scale = 1.0
        self.rect_start = None
        self.rect_end = None
        self.saved_rect = None     # QRect in image coords
        self._load_regions()
        self._setup_ui()

    def _setup_ui(self):
        layout = QVBoxLayout(self)

        btn_row = QHBoxLayout()
        self.save_btn = QPushButton("保存区域")
        self.save_btn.clicked.connect(self._save_regions)
        self.clear_btn = QPushButton("清除区域")
        self.clear_btn.clicked.connect(self._clear_region)
        self.status_label = QLabel("请点击缩略图选择样本页，然后拖拽画框")
        btn_row.addWidget(self.save_btn)
        btn_row.addWidget(self.clear_btn)
        btn_row.addWidget(self.status_label)
        btn_row.addStretch()
        layout.addLayout(btn_row)

        self.canvas = _Canvas(self)
        scroll = QScrollArea()
        scroll.setWidget(self.canvas)
        scroll.setWidgetResizable(False)
        layout.addWidget(scroll)

    def set_image(self, qimg: QImage):
        self.image = qimg
        self.canvas.set_image(qimg, self.saved_rect)

    def _load_regions(self):
        if os.path.exists(REGIONS_FILE):
            with open(REGIONS_FILE) as f:
                data = json.load(f)
            r = data.get("regions", [])
            if r:
                x, y, w, h = r[0]["rect"]
                self.saved_rect = QRect(x, y, w, h)
                return
        self.saved_rect = None

    def _save_regions(self):
        rect = self.canvas.current_rect
        if rect is None:
            return
        data = {
            "image_size": [1600, 900],
            "regions": [{"name": "browser", "rect": [rect.x(), rect.y(), rect.width(), rect.height()]}]
        }
        with open(REGIONS_FILE, "w") as f:
            json.dump(data, f, indent=2)
        self.saved_rect = rect
        self.status_label.setText(f"已保存: ({rect.x()},{rect.y()}) {rect.width()}×{rect.height()}")

    def _clear_region(self):
        self.canvas.current_rect = None
        self.canvas.update()
        self.saved_rect = None
        if os.path.exists(REGIONS_FILE):
            os.remove(REGIONS_FILE)
        self.status_label.setText("区域已清除")


class _Canvas(QWidget):
    def __init__(self, parent):
        super().__init__(parent)
        self.pixmap = None
        self.current_rect = None   # QRect in image coords
        self._drag_start = None
        self.scale = 1.0
        self.setCursor(Qt.CursorShape.CrossCursor)

    def set_image(self, qimg: QImage, saved_rect=None):
        self.pixmap = QPixmap.fromImage(qimg)
        self.scale = 1.0
        self.setFixedSize(self.pixmap.size())
        self.current_rect = saved_rect
        self.update()

    def paintEvent(self, event):
        if self.pixmap is None:
            return
        painter = QPainter(self)
        painter.drawPixmap(0, 0, self.pixmap)
        if self.current_rect:
            pen = QPen(QColor(255, 0, 0), 2)
            painter.setPen(pen)
            painter.drawRect(self.current_rect)

    def mousePressEvent(self, event):
        if event.button() == Qt.MouseButton.LeftButton:
            self._drag_start = event.pos()
            self.current_rect = None

    def mouseMoveEvent(self, event):
        if self._drag_start:
            self.current_rect = QRect(self._drag_start, event.pos()).normalized()
            self.update()

    def mouseReleaseEvent(self, event):
        if event.button() == Qt.MouseButton.LeftButton and self._drag_start:
            self.current_rect = QRect(self._drag_start, event.pos()).normalized()
            self._drag_start = None
            self.update()
```

### Task 5: Extractor (Claude haiku vision)

**Files:**
- Modify: `extractor.py`

**Step 1: Implement extractor**

```python
import os
import base64
import json
import fitz
from anthropic import Anthropic

client = Anthropic()  # reads ANTHROPIC_API_KEY from env

PROMPT = """This is a browser screenshot.
If the address bar URL contains "bilibili.com/video", extract the following and return ONLY valid JSON:
{
  "is_bilibili_video": true,
  "url": "<full URL from address bar>",
  "title": "<video title>",
  "views": "<view count as string, e.g. 12.3万>",
  "comments": "<comment count as string>",
  "date": "<publish date as string>"
}
If the URL does NOT contain "bilibili.com/video", return ONLY:
{"is_bilibili_video": false}
Return ONLY the JSON object, no markdown, no explanation."""


def extract_page(pdf_doc: fitz.Document, page_idx: int, crop_rect: dict | None) -> dict:
    """Extract metadata from one PDF page. Returns dict with fields or empty."""
    page = pdf_doc[page_idx]
    imgs = page.get_images(full=True)
    if not imgs:
        return {}

    xref = imgs[0][0]
    img_data = pdf_doc.extract_image(xref)
    image_bytes = img_data["image"]

    # Crop if region defined
    if crop_rect:
        from PIL import Image
        import io
        img = Image.open(io.BytesIO(image_bytes))
        x, y, w, h = crop_rect["rect"]
        img = img.crop((x, y, x + w, y + h))
        buf = io.BytesIO()
        img.save(buf, format="JPEG", quality=90)
        image_bytes = buf.getvalue()

    b64 = base64.standard_b64encode(image_bytes).decode()

    response = client.messages.create(
        model="claude-haiku-4-5-20251001",
        max_tokens=512,
        messages=[{
            "role": "user",
            "content": [
                {"type": "image", "source": {"type": "base64", "media_type": "image/jpeg", "data": b64}},
                {"type": "text", "text": PROMPT}
            ]
        }]
    )

    text = response.content[0].text.strip()
    try:
        return json.loads(text)
    except json.JSONDecodeError:
        # Try to extract JSON from response
        import re
        m = re.search(r'\{.*\}', text, re.DOTALL)
        if m:
            return json.loads(m.group())
        return {}
```

**Step 1: Implement exporter**

```python
from openpyxl import Workbook
from openpyxl.styles import Font, PatternFill, Alignment
from openpyxl.utils import get_column_letter

HEADERS = ["页码", "URL", "标题", "浏览次数", "评论次数", "发布时间"]

def export_to_excel(rows: list[dict], output_path: str):
    """
    rows: list of dicts with keys: page, url, title, views, comments, date
    """
    wb = Workbook()
    ws = wb.active
    ws.title = "Bilibili视频"

    # Header row
    header_fill = PatternFill("solid", fgColor="4472C4")
    header_font = Font(bold=True, color="FFFFFF")
    for col, h in enumerate(HEADERS, 1):
        cell = ws.cell(row=1, column=col, value=h)
        cell.fill = header_fill
        cell.font = header_font
        cell.alignment = Alignment(horizontal="center")

    # Data rows
    for row_idx, row in enumerate(rows, 2):
        ws.cell(row=row_idx, column=1, value=row.get("page", ""))
        ws.cell(row=row_idx, column=2, value=row.get("url", ""))
        ws.cell(row=row_idx, column=3, value=row.get("title", ""))
        ws.cell(row=row_idx, column=4, value=row.get("views", ""))
        ws.cell(row=row_idx, column=5, value=row.get("comments", ""))
        ws.cell(row=row_idx, column=6, value=row.get("date", ""))

    # Auto column width
    for col in range(1, len(HEADERS) + 1):
        max_len = max(
            len(str(ws.cell(row=r, column=col).value or ""))
            for r in range(1, ws.max_row + 1)
        )
        ws.column_dimensions[get_column_letter(col)].width = min(max_len + 4, 60)

    wb.save(output_path)
```

**Step 2: Add extraction logic**

```python
def _run_extraction(self):
    if not self.pdf_doc:
        QMessageBox.warning(self, "提示", "请先打开 PDF")
        return
    import json, os
    from extractor import extract_page
    crop_rect = None
    if os.path.exists("regions.json"):
        with open("regions.json") as f:
            data = json.load(f)
        regions = data.get("regions", [])
        if regions:
            crop_rect = regions[0]

    self.run_btn.setEnabled(False)
    self.results = []
    total = len(self.pdf_doc)
    self.progress.setMaximum(total)
    self.progress.setValue(0)
    self.table.setRowCount(0)

    def worker():
        from PyQt6.QtCore import QMetaObject, Q_ARG
        for i in range(total):
            result = extract_page(self.pdf_doc, i, crop_rect)
            if result.get("is_bilibili_video"):
                row = {
                    "page": i + 1,
                    "url": result.get("url", ""),
                    "title": result.get("title", ""),
                    "views": result.get("views", ""),
                    "comments": result.get("comments", ""),
                    "date": result.get("date", ""),
                }
                self.results.append(row)
                QMetaObject.invokeMethod(self, "_add_table_row",
                    Qt.ConnectionType.QueuedConnection,
                    Q_ARG(object, row))
            QMetaObject.invokeMethod(self, "_update_progress",
                Qt.ConnectionType.QueuedConnection,
                Q_ARG(int, i + 1))

        QMetaObject.invokeMethod(self, "_extraction_done",
            Qt.ConnectionType.QueuedConnection)

    threading.Thread(target=worker, daemon=True).start()

from PyQt6.QtCore import pyqtSlot

@pyqtSlot(object)
def _add_table_row(self, row):
    r = self.table.rowCount()
    self.table.insertRow(r)
    for col, key in enumerate(["page","url","title","views","comments","date"]):
        self.table.setItem(r, col, QTableWidgetItem(str(row.get(key, ""))))

@pyqtSlot(int)
def _update_progress(self, val):
    self.progress.setValue(val)

@pyqtSlot()
def _extraction_done(self):
    self.run_btn.setEnabled(True)
    self.export_btn.setEnabled(bool(self.results))

def _export_excel(self):
    from PyQt6.QtWidgets import QFileDialog
    from exporter import export_to_excel
    path, _ = QFileDialog.getSaveFileName(self, "保存 Excel", "output.xlsx", "Excel (*.xlsx)")
    if path:
        export_to_excel(self.results, path)
        QMessageBox.information(self, "完成", f"已导出到 {path}")
```
